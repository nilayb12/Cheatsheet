from pyspark.sql.functions import udtf

# One row in → many rows out
@udtf(returnType="word: string, length: int")
class SplitWords:
    def eval(self, text):
        for w in (text or "").split():
            yield (w, len(w))

# read

update, delete, upsert (merge)
dt = DeltaTable.forName(spark, "db.my_table")
dt = DeltaTable.forPath(spark, "/path/delta")
# update rows
dt.update(condition="id = 1",
          set={"status": "'active'"})
# delete rows
dt.delete("status = 'inactive'")
# merge / upsert
dt.alias("t").merge(
  source_df.alias("s"), "t.id = s.id")\
  .whenMatchedUpdateAll()\
  .whenNotMatchedInsertAll()\
  .execute()

Liquid clustering Databricks DBR 15.2+
CLUSTER BY
-- Replaces PARTITIONED BY + ZORDER
CREATE TABLE prod.sales.events (
  event_ts TIMESTAMP, user_id BIGINT
) USING DELTA
CLUSTER BY (event_ts, user_id);

-- Or let Databricks pick keys (UC managed)
CLUSTER BY AUTO;

-- Incremental; FULL to recluster (DBR 16.0+)
OPTIMIZE prod.sales.events FULL;
