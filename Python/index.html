# Library can sanitise before rendering
def html_render(tmpl: Template) -> str:
    import html
    parts = []; strings = iter(tmpl.strings)
    parts.append(next(strings))
    for i, s in zip(tmpl.interpolations, strings):
        parts.append(html.escape(str(i.value))); parts.append(s)
    return "".join(parts)

Dunder Methods
__repr__ · __add__ · __eq__/__hash__ · total_ordering
Core dunders
from functools import total_ordering
@total_ordering
class Money:
    def __init__(self, amt, cur="USD"):
        self.amount = round(amt,2); self.currency = cur
    def __repr__(self):
        return f"Money({self.amount!r})"
    def __add__(self, o):
        return Money(self.amount + o.amount, self.currency)
    def __eq__(self, o):
        return isinstance(o, Money) and self.amount==o.amount
    def __lt__(self, o): return self.amount < o.amount
    def __hash__(self): return hash(self.amount)

Decorators
@wraps · parametrised · @lru_cache · @cache
Function & parametrised
import functools, time
def log_calls(func):
    @functools.wraps(func)
    def wrapper(*args, **kw):
        print(f"Calling {func.__name__}")
        return func(*args, **kw)
    return wrapper
def retry(n=3, delay=1.0):
    def deco(func):
        @functools.wraps(func)
        def wrap(*a, **kw):
            for i in range(n):
                try: return func(*a, **kw)
                except:
                    if i==n-1: raise
                    time.sleep(delay)
        return wrap
    return deco

from pydantic import BaseModel, Field, field_validator, computed_field, ConfigDict
from typing import Annotated
PositiveInt = Annotated[int, Field(gt=0)]
class User(BaseModel):
    model_config = ConfigDict(str_strip_whitespace=True)
    id:    PositiveInt
    name:  str = Field(min_length=2, max_length=100)
    email: str = Field(pattern=r"^[\w.+-]+@[\w-]+\.[\w.]+$")
    age:   int = Field(ge=0, le=150)
    @field_validator("name")
    @classmethod
    def titlecase(cls, v): return v.title()
    @computed_field
    @property
    def is_adult(self) -> bool: return self.age >= 18

SQLAlchemy 2.0

Core & ORM
from sqlalchemy import create_engine, select, String, ForeignKey
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column, relationship, Session, selectinload
from typing import List
class Base(DeclarativeBase): pass
class Dept(Base):
    __tablename__ = "depts"
    id:   Mapped[int]      = mapped_column(primary_key=True)
    name: Mapped[str]      = mapped_column(String(100))
    emps: Mapped[List["Emp"]] = relationship(back_populates="dept")
class Emp(Base):
    __tablename__ = "emps"
    id:      Mapped[int] = mapped_column(primary_key=True)
    name:    Mapped[str] = mapped_column(String(100))
    dept_id: Mapped[int] = mapped_column(ForeignKey("depts.id"))
    dept:    Mapped["Dept"] = relationship(back_populates="emps")
engine = create_engine("sqlite:///app.db")
Base.metadata.create_all(engine)
with Session(engine) as s:
    eng = Dept(name="Engineering"); s.add(eng); s.flush()
    s.add(Emp(name="Alice", dept_id=eng.id)); s.commit()
    stmt = select(Dept).options(selectinload(Dept.emps))
    dept = s.execute(stmt).scalar_one()
    for e in dept.emps: print(e.name)
